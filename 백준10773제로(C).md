```c
#include <stdio.h>

int main()
{
    int max_size=100000;
    int n;
    int m;
    int arr1[max_size];
    int sum=0;
    int top = -1;
    
    printf("부를 숫자의 개수: ");
    scanf("%d",&n);
    for (int i=0; i<n; i++) {
        scanf("%d",&m);
        
        if (m==0) {
            if (top>=0) {
                sum-=arr1[top];
                top--;
            }
        }else {
            top++;
            arr1[top]=m;
            sum+=m;
        }
        
    }
    printf("결과값: %d", sum);

    return 0;
}
```
